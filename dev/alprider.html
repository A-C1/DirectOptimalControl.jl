function dyn(x, u, t, p)
    x1 = -10x[1] + u[1] + u[2]
    x2 = -2x[2] + u[1] + 2u[2]
    x3 = -3x[3] + 5x[4] + u[1] - u[2]
    x4 = 5x[3] - 3x[4] + u[1] + 3u[2]

    return [x1, x2, x3, x4]
end
